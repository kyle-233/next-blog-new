## curry()

implement `curry()`

``` js
const curry = (fn) => {
  return function curried(...args) {
    if (args.length >= fn.length) {
      fn(...args)
    } else {
      return (...newArgs) => {
        return curried(...args, ...newArgs)
      }
    }
  }
}
```

## curry() with placeholder

implement `curry()` which also supports placeholder.

```js
const curry = (fn) => {
  return function curried(...args) {
    if (args.length >= fn.length && !args.includes(curry.placeholder)) {
      fn(...args)
    } else {
      return (...newArgs) => {
        return curried(...(args.map(arg => arg === curry.placeholder ? newArgs.shift() : arg)), ...newArgs)
      }
    }
  }
}
curry.placeholder = Symbol()
```
